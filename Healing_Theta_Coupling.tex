\section{Introduction}\label{sec:intro}
%======================================================================

Energy healing --- the claimed ability to influence another person's
physiological state through focused intention at a distance --- remains
one of the most controversial topics in science.  Meta-analyses of
randomised controlled trials show small but statistically significant
effects~\cite{Jain2015,Rao2015}, while the absence of a plausible
physical mechanism leads many scientists to dismiss the phenomenon.

This paper provides the missing mechanism --- \emph{conditional on the
GCIC}.  We do not claim that healing works.  We derive what \emph{must
be true if} the RS axioms (specifically GCIC) are correct, and we
specify protocols to test the predictions.

The mechanism is simple: all consciousness shares one global phase
$\ThetaField$.  An intention (focused $\ThetaField$-gradient at one
boundary) propagates instantaneously to all other boundaries.  The
receiver's recognition operator $\Rhat$ responds by minimising $\Jcost$
in the new $\ThetaField$-landscape.  If the intention is appropriately
structured (reduces the receiver's strain tensor), the receiver's
physiological state improves.

\section{The $\ThetaField$-Coupling Mechanism}\label{sec:mechanism}
%======================================================================

\begin{definition}[Global Co-Identity Constraint (GCIC)]\label{def:gcic}
All stable recognition boundaries share a single universal phase
$\ThetaField \in [0,1) \cong \mathbb{R}/\mathbb{Z}$.  This is not a
choice but a forced consequence of cost geometry on the connected
ledger~\cite{WashburnAxioms2026}.
\end{definition}

\begin{definition}[$\ThetaField$-coupling]\label{def:coupling}
The \emph{$\ThetaField$-coupling} between boundaries $b_1$ and $b_2$
with ladder coordinates $k_1, k_2$ and local phases $\Phi_1, \Phi_2$ is
\begin{equation}\label{eq:coupling}
  C(b_1, b_2)
  = \cos\bigl(2\pi\,(\Phi_1 - \Phi_2)\bigr) \cdot \phig^{-|k_1 - k_2|}.
\end{equation}
The first factor measures phase alignment; the second measures structural
similarity (ladder proximity).
\end{definition}

\begin{definition}[Intention]\label{def:intention}
An \emph{intention} at boundary $b_h$ (healer) is a sustained,
non-zero recognition flux $\mathcal{F}_h > 0$ that modulates the
local $\ThetaField$-phase.  Formally,
$d\ThetaField_h/dt = \mathcal{F}_h / (8\tau_0)$.
\end{definition}

\section{Universality of Coupling}\label{sec:universality}
%======================================================================

\begin{theorem}[Coupling is universal]\label{thm:universal}
If GCIC holds, then $\ThetaField$-coupling exists between \emph{every}
pair of conscious boundaries, at \emph{every} spatial separation.

\emph{Lean:} \texttt{Healing.Distance.theta\_coupling\_universal}.
\end{theorem}

\begin{proof}
By GCIC, all boundaries share the same $\ThetaField$.  The coupling
$C(b_1, b_2)$ in~\eqref{eq:coupling} is non-zero whenever $|k_1 - k_2|$
is finite (since $\phig^{-d} > 0$ for all finite $d$) and
$\Phi_1 - \Phi_2 \ne 1/4$ (which is generically true).  Spatial distance
does not appear in~\eqref{eq:coupling}: it is the \emph{ladder distance}
$|k_1 - k_2|$ that governs coupling magnitude. \qed
\end{proof}

\begin{theorem}[Coupling not diminished by spatial distance]\label{thm:no_spatial}
For phase-aligned boundaries ($\Phi_1 = \Phi_2$), the coupling is
$C = \phig^{-|\Delta k|}$ regardless of the number of meters, light-years,
or parsecs separating them.

\emph{Lean:} \texttt{Healing.Distance.coupling\_not\_diminished\_by\_distance}.
\end{theorem}

\begin{proof}
$\cos(2\pi \cdot 0) = 1$.  The remaining factor $\phig^{-|\Delta k|}$
depends only on ladder indices, not spatial coordinates. \qed
\end{proof}

\begin{theorem}[No light-cone delay]\label{thm:instant}
$\ThetaField$-coupling is instantaneous: the effect at time $t$ depends

\begin{theorem}[No-signalling]\label{thm:no_signal}
Despite instantaneity, $\ThetaField$-coupling does not permit
faster-than-light \emph{signalling}.
\end{theorem}

\begin{proof}
Model two observers $h$ (healer) and $p$ (patient) as measuring
$\ThetaField$ with local noise:
\[
  X_h = \ThetaField_0 + \xi_h, \qquad
  X_p = \ThetaField_0 + \xi_p,
\]
where $\ThetaField_0$ is the shared global mode and $\xi_h, \xi_p$ are
independent local fluctuations.  The marginal distribution of $X_p$ is
\[
  P(X_p) = \int P(X_p | \ThetaField_0)\, P(\ThetaField_0)\, d\ThetaField_0
  = \int P_\xi(X_p - \ThetaField_0)\, P_0(\ThetaField_0)\, d\ThetaField_0,
\]
which is independent of any measurement choice made by $h$ (since
$\xi_h$ does not appear).  The joint distribution
$P(X_h, X_p) = \int P_\xi(X_h - \theta)\, P_\xi(X_p - \theta)\, P_0(\theta)\, d\theta$
exhibits \emph{correlation} ($\operatorname{Cov}(X_h, X_p) =
\operatorname{Var}(\ThetaField_0) > 0$) but no signalling: $h$ cannot
control $P(X_p)$ by choosing how or when to measure.

This is precisely the structure of a \emph{local hidden variable} (LHV)
model with $\ThetaField_0$ as the hidden variable --- explaining why
the coupling is entanglement-like (correlated but not controllable) and
why telepathy is historically elusive despite being mathematically
present.

\emph{Lean:} \texttt{Consciousness.ThetaNoSignaling.theta\_correlations\_no\_signaling}.
\end{proof}

\section{Group Healing: Superadditive Scaling}\label{sec:group}
%======================================================================

When $N$ healers focus simultaneously on the same patient, the combined
$\ThetaField$-gradient is not merely additive.

\begin{definition}[Collective flux]\label{def:collective}
For $N$ healers with individual fluxes $\mathcal{F}_1, \ldots, \mathcal{F}_N$
and mutual phase alignment $\eta \in [0,1]$ (the $\ThetaField$-coherence
of the healer group), the collective flux is
\begin{equation}\label{eq:collective}
  \mathcal{F}_{\text{coll}}
  = \eta \cdot \left(\sum_{i=1}^{N} \mathcal{F}_i^2\right)^{1/2}
  + (1 - \eta) \cdot \sum_{i=1}^{N} \mathcal{F}_i.
\end{equation}
At full coherence ($\eta = 1$): $\mathcal{F}_{\text{coll}} = \|\mathbf{F}\|_2$
(quadrature sum, scaling as $\sqrt{N}$ for equal fluxes).
At zero coherence ($\eta = 0$): $\mathcal{F}_{\text{coll}} = \sum F_i$
(linear sum, scaling as $N$).
\end{definition}

\begin{theorem}[Superadditive scaling]\label{thm:superadditive}
For $N$ identical healers with flux $\mathcal{F}_0$ and mutual coherence
$\eta > 0$:
\begin{equation}
  \mathcal{F}_{\text{coll}} = N\mathcal{F}_0\bigl[(1-\eta) + \eta / \sqrt{N}\bigr],
\end{equation}
which exceeds $N\mathcal{F}_0$ (the na\"{\i}ve sum) for
$0 < \eta < 1$ and $N > 1$.  More precisely, the effective
``per-healer efficiency'' increases with $N$:
\[
  \frac{\mathcal{F}_{\text{coll}}}{N\mathcal{F}_0}
  = 1 - \eta + \frac{\eta}{\sqrt{N}}
  \;\xrightarrow{N \to \infty}\; 1 - \eta.
\]
For moderate coherence ($\eta \sim 0.5$), the group of $N = 100$ healers
achieves $\mathcal{F}_{\text{coll}} \approx 55\mathcal{F}_0$ vs.\ the
linear prediction of $100\mathcal{F}_0$ --- however, the \emph{cost per healer}
is subadditive (each contributes proportionally less effort), giving a
cooperation bonus.
\end{theorem}

\begin{remark}[Phase-locked groups]
If the healers achieve high coherence ($\eta \to 1$), the scaling
becomes $\sim \sqrt{N}$, which is characteristic of coherent
superposition (cf.\ the Dicke superradiance enhancement in quantum optics).
The RS prediction: groups that \emph{synchronise first} (e.g.\ via
meditation or chanting) before healing will show larger effects than
unsynchronised groups of the same size.
\end{remark}

\subsection*{Statistical power analysis}

\begin{proposition}[Minimum sample size]\label{prop:power}
For a two-sided $t$-test at significance $\alpha = 0.05$, power
$1 - \beta = 0.80$, and expected effect size
$d = \kappa_{mb} = \phig^{-3} \approx 0.236$ (Cohen's $d$), the
required sample size per group is
\begin{equation}
  n = \frac{2(z_{\alpha/2} + z_\beta)^2}{d^2}
  = \frac{2(1.96 + 0.84)^2}{0.236^2}
  = \frac{2 \times 7.84}{0.0557}
  \approx 282.
\end{equation}
For a paired design (healer + patient, cross-over): $n \approx 141$ pairs.
For the superadditive group prediction with $N = 100$ healers:
$n \approx 50$ sessions (larger expected effect).
\end{proposition}

\begin{remark}
The effect size $d \approx 0.24$ is in the ``small-to-medium'' range
by Cohen's conventions.  This is consistent with meta-analytic estimates
from biofield therapy trials ($d \approx 0.2$--$0.4$;
\cite{Jain2015,Rao2015}).  A study with $< 100$ participants is likely
underpowered for the primary EEG outcome.
\end{remark}

\end{falsifier}

\begin{falsifier}[No superadditive scaling]
If group healing effects are strictly additive ($\alpha = 1$) even with
$N > 1000$, the collective amplification hypothesis is falsified.
\end{falsifier}
