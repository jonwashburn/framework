	\section{Schur/Herglotz pinch: a minor but necessary condition}
	
	\begin{lemma}[Removable singularity under Schur bound]\label{lem:removable}
		Let $D\subset\C$ be a disc centered at $\rho$ and let $\Theta$ be holomorphic on $D\setminus\{\rho\}$ with $|\Theta|<1$ there.
		Then $\Theta$ extends holomorphically to $D$.
	\end{lemma}
	\begin{proof}
		Since $\Theta$ is bounded on $D\setminus\{\rho\}$, Riemann's removable singularity theorem applies.
	\end{proof}
	
	\begin{corollary}[Schur bound prevents poles of the Cayley inverse]\label{cor:pinch}
		Let $U$ be a domain and suppose $\Theta$ is meromorphic on $U$ and satisfies $|\Theta|\le 1$ on $U$ away from its poles.
		\Rev{Assume additionally that $\Theta$ is not identically $1$ on any connected component of $U$.}
		Then $\Theta$ extends holomorphically to $U$ and satisfies $|\Theta|\le 1$ on $U$.
		Moreover the Cayley inverse $(1+\Theta)/(1-\Theta)$ is holomorphic on $U$ and has nonnegative real part there.
	\end{corollary}
	\begin{proof}
		Poles of a meromorphic function are isolated, hence locally the hypothesis reduces to Lemma~\ref{lem:removable} (if $|\Theta|<1$).
		If $\Theta$ is holomorphic and $|\Theta|\le 1$ on a connected domain, either $|\Theta|<1$ everywhere or $\Theta$ is constant of modulus $1$
		(by the maximum modulus principle).
		\Rev{The added condition rules out $\Theta\equiv 1$, so on each component one has $|\Theta|<1$ everywhere.}
		Then the Cayley inverse is holomorphic and maps the unit disk into the right half-plane.
	\end{proof}
	
	\section{What is missing in the finite-to-infinite Pick step}
	
	\subsection{A correct (and standard) operator lemma you \emph{can} use}
	
	\begin{lemma}[Schur complement criterion for block operators]\label{lem:schur-complement}
		Let $\mathcal H=\mathcal H_1\oplus\mathcal H_2$ and let
		\[
		P \;=\;
		\begin{pmatrix}
			A & B\\
			B^* & D
		\end{pmatrix}
		\]
		be a bounded self-adjoint operator on $\mathcal H$.
		Assume $D$ is invertible and $D\succeq \gamma I_{\mathcal H_2}$ for some $\gamma>0$.
		If
		\[
		A - B\,D^{-1}B^* \succeq 0 \quad \text{on }\mathcal H_1,
		\]
		then $P\succeq 0$ on $\mathcal H$.
	\end{lemma}
	\begin{proof}
		For $x\in\mathcal H_1$, $y\in\mathcal H_2$ set $y'=y+D^{-1}B^*x$.
		Then
		\[
		\langle P(x\oplus y),x\oplus y\rangle
		= \langle (A-BD^{-1}B^*)x,x\rangle + \langle Dy',y'\rangle \ge 0.
		\]
	\end{proof}

		Fix $N\ge1$ and decompose $\ell^2(\N_0)=\mathrm{span}\{e_0,\dots,e_{N-1}\}\oplus(\cdot)^\perp$.
		Write the corresponding block operator as
		\[
		P(\theta)=
		\begin{pmatrix}
			A & B\\
			B^* & D
		\end{pmatrix}.
		\]
		
		\Rev{To conclude $P(\theta)\succeq 0$ from finite data, your proof must provide \textbf{one} of the following:}
		
		\begin{itemize}
			\item \Rev{\textbf{(A) Direct certificate:} a rigorous argument (analytic or machine-checked) that $P(\theta)$ is positive semidefinite as an operator on $\ell^2(\N_0)$.}
			\item \Rev{\textbf{(B) Block bounds:} explicit numbers $\delta,\beta,\gamma>0$ such that

		\end{itemize}
		
		\Rev{Important: a bound of the form $\sum_{n\ge N}(n+1)|a_n|^2\le \varepsilon_N^2$ does \emph{not}, by itself, yield the needed operator-norm bounds on $B$ and $D$ unless a fully written analytic estimate is supplied.  Your current text sketches such a passage, but it is not rigorous as stated and must be replaced by a complete derivation with explicit constants, or else the verifier must output the required block bounds directly.}
	\end{hypothesis}
