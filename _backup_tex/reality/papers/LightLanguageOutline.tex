\section*{Working Outline}

\begin{enumerate}
  \item \textbf{Executive Overview}
  \begin{itemize}
    \item Motivation for a zero-parameter semantic language.
    \item Summary of current achievements (20-token dictionary, $\varphi$-lattice alignment, 42/42 tests, Lean proofs).
    \item Scope of this paper and relation to existing reports.
  \end{itemize}

  \item \textbf{Recognition Science Foundations}
  \begin{itemize}
    \item RS axioms and gates (recognition scale, neutrality, ledger parity).
    \item Derivation of the eight-beat frame and neutral subspace.
    \item Statement of exclusivity theorems (Lean references).
  \end{itemize}

  \item \textbf{Universal Light Language Specification}
  \begin{itemize}
    \item Definition of ledger structures and LNAL operators (LISTEN/LOCK/BALANCE/FOLD/BRAID).
    \item Coercivity and neutrality invariants.
    \item Operator algebra and equivalence under cyclic phase.
  \end{itemize}

  \item \textbf{CPM Discovery Pipeline}
  \begin{itemize}
    \item Construction of recognition suites (14 moves $\times$ style variants).
    \item Minimum Description Length objective; token acceptance and pruning.
    \item Stability across seeds and style variants; MDL margins.
  \end{itemize}

  \item \textbf{$\varphi$-Lattice Structure}
  \begin{itemize}
    \item Pairwise distance assignment to $\varphi^k$ lattice.
    \item Bootstrap and negative-control analyses.
    \item Interpretation of residual statistics and p-value thresholds.
  \end{itemize}

  \item \textbf{Grammar and Motifs}
  \begin{itemize}
    \item Proof obligations for neutrality and ledger parity in sequences.
    \item Motif mining pipeline; distance-band diversity.
    \item Lean status of grammar lemmas.
  \end{itemize}

  \item \textbf{Truthification Pipeline}
  \begin{itemize}
    \item Truthify CLI workflow (normal form extraction, perturbation stability, legality checks).
    \item Certificate structure (JSON schema, metadata, margins).
    \item Lean export stubs and roadmap for mechanised exclusivity proofs.
  \end{itemize}

  \item \textbf{Empirical Validation}
  \begin{itemize}
    \item Dictionary metrics (token count, MDL gap, reconstruction).
    \item Cross-modal persistence and drift statistics (seed table).
    \item Benchmark examples with truth certificates.
  \end{itemize}

  \item \textbf{Towards the Singular Language}
  \begin{itemize}
    \item Remaining proof obligations (global exclusivity, counterfactual elimination).
    \item Integration plan for large-scale truthify runs and Lean certification.
    \item Deployment considerations (API, audit bundles, reproducibility).
  \end{itemize}

  \item \textbf{Appendices}
  \begin{itemize}
    \item Certificate JSON schema and example.
    \item Snippets from Lean exports.
    \item Command sequences for reproducing the metrics.
  \end{itemize}
\end{enumerate}
