\title[RM2/RM2U reduction note]{RM2/RM2U reduction note: the log-critical $\ell=2$ tail moment and a single global tightness gate}
\author{Jonathan Washburn}

\begin{document}
\maketitle

\begin{abstract}
This note extracts a referee-checkable core from the running-max blow-up program for 3D incompressible Navier--Stokes.
The fixed-frame compactness obstruction (RM2) is identified with a \emph{log-critical $\ell=2$ tail moment} of vorticity:
the tail vorticity induces an affine harmonic velocity mode whose symmetric gradient at the origin is an explicit $\ell=2$ shell moment.
We record a coercive $\ell=2$ estimate (RM2U) that suffices to bound this tail moment, and we isolate a single global tightness condition
(\emph{uniform exterior weighted enstrophy}, UEWE) that would imply the coercive estimate.
At present the only missing analytic input is a genuine blow-up-variable tightness/RTD/UEWE mechanism; everything else in this note is unconditional.
\end{abstract}

\section{The RM2 obstruction as a log-critical $\ell=2$ tail moment}

\subsection{Tail strain at the origin (explicit formula)}

\begin{lemma}[Biot--Savart tail strain formula (explicit $\ell=2$ moment)]\label{lem:tail-strain-formula}
Let $\Omega:\R^3\to\R^3$ be smooth and supported in $\{|w|>1\}$, and define its Biot--Savart velocity
\[
u(x)=\frac{1}{4\pi}\int_{\R^3}\frac{(x-w)\times \Omega(w)}{|x-w|^3}\,dw.
\]
Then $u$ is smooth and harmonic on $B_1(0)$ and divergence-free on $\R^3$.
Moreover the symmetric gradient at the origin,
\[
S(0):=\tfrac12\bigl(\nabla u(0)+\nabla u(0)^T\bigr)\in\R^{3\times 3}_{\mathrm{sym}},
\]
is given by the explicit moment identity
\[
S(0)
=-\frac{3}{8\pi}\int_{|w|>1}\frac{(w\times\Omega(w))\otimes w+w\otimes(w\times\Omega(w))}{|w|^5}\,dw.
\]
In particular $\operatorname{tr}S(0)=0$.
Equivalently, for every unit vector $b\in\Sbb^2$,
\[
b\cdot S(0)\,b
=-\frac{3}{4\pi}\int_{|w|>1}\frac{(b\cdot w)\,\bigl((b\times w)\cdot\Omega(w)\bigr)}{|w|^5}\,dw
=-\frac{3}{4\pi}\int_{1}^{\infty}\frac{dr}{r}\int_{\Sbb^2}(b\cdot\theta)\,\bigl((b\times\theta)\cdot\Omega(r\theta)\bigr)\,d\theta.
\]
\end{lemma}

\begin{proof}
Since $\Omega$ is supported in $\{|w|>1\}$, the kernel $x\mapsto (x-w)/|x-w|^3$ is harmonic on $B_1(0)$ for each fixed $w$ in the support, so $u$ is harmonic on $B_1(0)$ and smooth there.
Differentiate under the integral sign (justified by smoothness and the separation of the support from $B_1$), write $r=x-w$ and $f(r):=r/|r|^3$ so that $u(x)=\frac{1}{4\pi}\int f(x-w)\times \Omega(w)\,dw$.
For $i,k\in\{1,2,3\}$,
\[
\partial_{x_i} f_k(r)=\partial_{r_i}\bigl(r_k|r|^{-3}\bigr)=\delta_{ik}|r|^{-3}-3\,r_i r_k|r|^{-5}.
\]
Hence at $x=0$ (so $r=-w$),
\[
\partial_{x_i} u(0)=\frac{1}{4\pi}\int \bigl(\partial_{x_i} f(-w)\bigr)\times \Omega(w)\,dw.
\]
When we take the symmetric part $\tfrac12(\nabla u(0)+\nabla u(0)^T)$, the $\delta_{ik}|w|^{-3}$ contribution cancels, leaving the stated identity.
Taking the trace gives $\operatorname{tr}S(0)\propto \int (w\times\Omega)\cdot w\,|w|^{-5}\,dw=0$.
The directional formula follows by contracting with $b\otimes b$ and observing $b\cdot(w\times\Omega)=(b\times w)\cdot\Omega$, followed by the change of variables $w=r\theta$.
\end{proof}

\subsection{RM2 in one line}

\begin{corollary}[RM2 $\iff$ bounded $\ell=2$ tail moment (informal equivalence)]\label{cor:RM2-equivalence}
In the running-max extraction, the fixed-frame compactness obstruction (RM2) is generated by the affine/harmonic velocity mode induced by the far field.
Lemma~\ref{lem:tail-strain-formula} identifies the corresponding coefficient at the blow-up center as a borderline log-shell $\ell=2$ tail moment.
Accordingly, any RM2 closure must control (uniformly or integrably, along the blow-up sequence) the induced tail strain matrix $S(0,t)$.
\end{corollary}

\begin{remark}
Corollary~\ref{cor:RM2-equivalence} is an equivalence ``at the level of the program'' (it is not a new PDE theorem by itself): it pinpoints \emph{what} must be controlled to make fixed-frame limits exist.
\end{remark}

\section{A single global gate that would close RM2 (RM2U via UEWE)}

\subsection{The RM2U target (coercive $\ell=2$ control)}

\begin{definition}[The transverse $\ell=2$ coefficient]
Fix $b\in\Sbb^2$ and define the tangential test field on $\Sbb^2$ by
\[
\Phi_b(\theta):=(b\cdot \theta)\,(\theta\times b),
\qquad \theta\in\Sbb^2.
\]
For a smooth vorticity field $\omega(x,t)$ define
\[
A_b(r,t):=\int_{\Sbb^2}\omega(r\theta,t)\cdot \Phi_b(\theta)\,d\theta.
\]
\end{definition}

\begin{theorem}[RM2U target: coercive $\ell=2$ tail control]\label{thm:RM2U-target}
\textbf{[OPEN TARGET / GLOBAL GATE.]}
For the running-max/vorticity-normalized ancient element $\omega^\infty=\curl u^\infty$, show there exists $K<\infty$ such that for all $t\le 0$,
\[
\sup_{b\in\Sbb^2}\left(\int_{1}^{\infty}\bigl|(\partial_r A_b^\infty)(r,t)\bigr|^2\,r^2\,dr\ +\ \int_{1}^{\infty}\bigl|A_b^\infty(r,t)\bigr|^2\,dr\right)\ \le\ K.
\]
\end{theorem}

\subsection{A clean sufficient condition: UEWE}

\begin{theorem}[UEWE target: uniform exterior weighted enstrophy]\label{thm:RM2U-weighted-enstrophy-target}
\textbf{[OPEN TARGET / GLOBAL GATE.]}
For the running-max/vorticity-normalized ancient element $\omega^\infty=\curl u^\infty$, show there exists $M<\infty$ such that for all $t\le 0$,
\[
\int_{|x|\ge 1}\left(\frac{|\omega^\infty(x,t)|^2}{|x|^2}+|\nabla \omega^\infty(x,t)|^2\right)\,dx\ \le\ M.
\]
\end{theorem}

\begin{lemma}[UEWE $\Rightarrow$ coercive $\ell=2$ control]\label{lem:RM2U-weighted-sufficient}
Fix a time $t$ and let $\omega(\cdot,t)$ be smooth.
If
\[
\int_{|x|\ge 1}\left(\frac{|\omega(x,t)|^2}{|x|^2}+|\nabla \omega(x,t)|^2\right)\,dx\ \le\ M,
\]
then for every $b\in\Sbb^2$ the corresponding coefficient satisfies
\[
\int_{1}^{\infty}\bigl|(\partial_r A_b)(r,t)\bigr|^2\,r^2\,dr\ +\ \int_{1}^{\infty}\bigl|A_b(r,t)\bigr|^2\,dr
\ \le\ C\,M,
\]
with a universal constant $C$ (depending only on $\|\Phi_b\|_{L^2(\Sbb^2)}^2$, hence independent of $b$).
\end{lemma}

\begin{proof}
By Cauchy--Schwarz on $\Sbb^2$,
\[
|A_b(r,t)|
=\left|\int_{\Sbb^2}\omega(r\theta,t)\cdot\Phi_b(\theta)\,d\theta\right|
\le \|\omega(r\cdot,t)\|_{L^2(\Sbb^2)}\,\|\Phi_b\|_{L^2(\Sbb^2)}.
\]
Squaring and integrating in $r\in[1,\infty)$ gives
\[
\int_1^\infty |A_b(r,t)|^2\,dr
\le \|\Phi_b\|_{L^2(\Sbb^2)}^2\int_{|x|\ge 1}\frac{|\omega(x,t)|^2}{|x|^2}\,dx.
\]
Similarly,
\[
(\partial_r A_b)(r,t)=\int_{\Sbb^2}(\partial_r\omega)(r\theta,t)\cdot\Phi_b(\theta)\,d\theta,
\]
so the same Cauchy--Schwarz bound yields
\[
\int_1^\infty |(\partial_r A_b)(r,t)|^2\,r^2\,dr
\le \|\Phi_b\|_{L^2(\Sbb^2)}^2\int_{|x|\ge 1}|\partial_r\omega(x,t)|^2\,dx
\le \|\Phi_b\|_{L^2(\Sbb^2)}^2\int_{|x|\ge 1}|\nabla\omega(x,t)|^2\,dx.
\]
Adding the two bounds gives the claim with $C=\|\Phi_b\|_{L^2(\Sbb^2)}^2$ (uniform in $b$ by rotational invariance).
\end{proof}

\subsection{Coercive $\ell=2$ control bounds the RM2 tail moment}

\begin{theorem}[Coercive $\ell=2$ control $\Rightarrow$ bounded tail moment]\label{thm:RM2U-closure-from-coercive}
Assume the RM2U coercive bound from Theorem~\ref{thm:RM2U-target} holds with constant $K$ (uniformly for all $t\le 0$).
Then for every $t\le 0$ the log-critical shell moments
\[
\Sigma_b^{1,\infty}(t):=\int_{1}^{\infty}\frac{A_b^\infty(r,t)}{r}\,dr
\]
converge absolutely and satisfy $\sup_{b\in\Sbb^2}|\Sigma_b^{1,\infty}(t)|\le K^{1/2}$.
In particular, the induced tail strain matrix $S(0,t)$ (as in Lemma~\ref{lem:tail-strain-formula}) is uniformly bounded in $t\le 0$.
\end{theorem}

\begin{proof}
Fix $b$ and apply Cauchy--Schwarz:
\[
\int_{1}^{\infty}\frac{|A_b^\infty(r,t)|}{r}\,dr
\le \left(\int_{1}^{\infty}|A_b^\infty(r,t)|^2\,dr\right)^{1/2}
\left(\int_{1}^{\infty}\frac{dr}{r^2}\right)^{1/2}
\le K^{1/2}.
\]
This gives absolute convergence and the uniform bound on $\Sigma_b^{1,\infty}$.
The tail strain bound then follows by the directional identity in Lemma~\ref{lem:tail-strain-formula}:
for each $b$, $b\cdot S(0,t)b$ is proportional to $\Sigma_b^{1,\infty}(t)$.
\end{proof}

\section{The single missing step (open problem)}

\begin{remark}[Single global bottleneck]
All implications in this note are elementary once the correct coefficient/moment is identified.
The only missing analytic input needed to close RM2 via RM2U is a \emph{genuine global tightness} estimate for the running-max ancient element,
such as UEWE (Theorem~\ref{thm:RM2U-weighted-enstrophy-target}) or an equivalent relative tail depletion (RTD) statement in blow-up variables.
\end{remark}
