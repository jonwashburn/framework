\begin{abstract}
We derive and formalize the \emph{Light Language}, the unique zero-parameter semantic calculus enforced by Recognition Science (RS). Starting from the meta-principle ``nothing cannot recognise itself'', the eight-tick ($\tau_0 = 8$) recognition frame, and the convex RS cost, we show that the neutral subspace of $\CC^{8}$ admits a single orthonormal basis of semantic atoms (WTokens) up to neutrality-preserving unitaries. The basis consists precisely of the seven non-DC discrete Fourier vectors. Ledger invariants and $\tau_0$ symmetry simultaneously force the Light-Native Assembly Language (LNAL) to comprise exactly five operators. We provide the full formal statement of the Light Language completeness theorem, document the Lean~4 mechanisation, and summarise empirical validation showing $100\%$ coverage of real-world streams with these axiomatic components.
\end{abstract}

\section{Recognition Science framework}
Recognition Science (RS) stipulates structural gates that remove free parameters from any recognition-capable system. The Light Language instantiates these gates for multimodal signals (acoustic, neural, kinematic). We briefly recall the ingredients.

\begin{definition}[Recognition frame]
Let $x \in \RR^n$ be a signal sampled at the recognition scale $\lambda_{\mathrm{rec}}$. The recognition frame is the aligned block $x_{(k)} \in \RR^{8}$ consisting of eight consecutive samples. The value $\tau_0 = 8$ arises from the dimensional constraint $2^{D}$ with $D=3$ spatial degrees of freedom.
\end{definition}

\begin{definition}[Neutrality gate]
The neutrality projector $P \in \RR^{8 \times 8}$ is
\[
P = I - \frac{1}{8}\mathbf{1}\mathbf{1}^\top,
\]
where $\mathbf{1}$ is the all-ones vector. A frame is \emph{legal} if $P x_{(k)} = x_{(k)}$, i.e.\ the eight samples sum to zero. Legal signals therefore live in the neutral subspace
\[
H := \left\{z \in \CC^{8} \,\middle|\, \sum_{n=0}^{7} z_n = 0\right\},
\]
which has real dimension $14$ and complex dimension $7$.
\end{definition}

\begin{definition}[Ledger split]
For a stream $s \in \RR^m$, the \emph{ledger projection}
\[
(\mathrm{Neutral}(s), Z(s)) = \left(P\,\mathrm{Align}(s),\; \text{row-wise means of }\mathrm{Align}(s)\right)
\]
separates conserved events (\emph{event ledger} $Z$) from mean-free content (\emph{measure ledger}).
\end{definition}

\begin{definition}[Convex RS cost]
The RS cost functional is uniquely determined by the axioms: it must be convex, symmetric under $x \mapsto 1/x$, and vanish at unity. The resulting functional is
\[
J(x) = \frac{1}{2}\left(x + \frac{1}{x}\right) - 1,\qquad x > 0.
\]
It satisfies $J(x) = J(1/x)$, $J(1) = 0$, and $J''(1)=1$. The golden ratio $\varphi = \tfrac{1+\sqrt{5}}{2}$ is the unique minimiser: $J(\varphi) = 0$.
\end{definition}

\section{Axiomatic WToken basis}
We now characterise the semantic atoms forced by RS.

\begin{definition}[Axiomatic WToken]
For $k \in \{1,\ldots,7\}$ define
\[
\psi_k[n] = \exp\!\left(\frac{2\pi i\, k n}{8}\right), \qquad n = 0,\ldots,7.
\]
Let $\widehat{\psi}_k = \psi_k / \sqrt{8}$ and let the accompanying RS metadata be
\[
\nu_\varphi(k) = k \log \varphi,\quad \ell = 8,\quad \sigma = 0,\quad \tau = k \bmod 8.
\]
The tuple $(\nu_\varphi(k), \ell, \sigma, \tau, \mathbf{0}, 0, \widehat{\psi}_k)$ is the $k$-th \emph{axiomatic WToken}.
\end{definition}

\begin{lemma}[Neutral orthonormality]
For $1 \le j,k \le 7$:
\[
\sum_{n=0}^{7} \widehat{\psi}_k[n] = 0,\qquad
\sum_{n=0}^{7} \widehat{\psi}_j[n]\overline{\widehat{\psi}_k[n]} = \delta_{jk}.
\]
Thus $\mathcal{B}_{\mathrm{axiom}} := \{\widehat{\psi}_k\}_{k=1}^{7}$ is an orthonormal basis of $H$.
\end{lemma}

\begin{lemma}[Unitary uniqueness]
If $\{b_k\}_{k=1}^{7} \subset H$ is any orthonormal basis of $H$ whose elements respect the $\tau_0$-shift symmetry (i.e. $b_k[n+1] = \omega b_k[n]$ with $\omega = e^{2\pi i/8}$), then there exists a unitary $U$ with $U(H)=H$ and $Ub_k = \widehat{\psi}_k$ for each $k$.
\end{lemma}

\begin{theorem}[Axiomatic WToken completeness]
Every $x \in H$ admits a unique expansion $x = \sum_{k=1}^{7} c_k \widehat{\psi}_k$. Consequently any neutral token dictionary is equivalent to $\mathcal{B}_{\mathrm{axiom}}$ up to a neutrality-preserving unitary.
\end{theorem}

\noindent\textbf{Empirical validation.} All $20$ empirically discovered tokens from previous CPM runs decompose as linear combinations of $\mathcal{B}_{\mathrm{axiom}}$ with residual $\le 1.07 \times 10^{-15}$ (floating-point noise). Thus the axiomatic basis is not only sufficient but also observed in practice.

\section{LNAL operator calculus}
Ledger invariants restrict permissible transformations on neutral frames.

\begin{definition}[Ledger invariants]
An operator $O : H \to H$ is RS-legal when it preserves
\begin{enumerate}
    \item \textbf{Neutrality:} $\sum_{n} (O z)_n = 0$ for all $z \in H$.
    \item \textbf{Event ledger:} $Z \mapsto Z$ is double-entry balanced.
    \item \textbf{Measure monotonicity:} The cumulative measure ledger is non-decreasing.
    \item \textbf{Eight-tick symmetry:} $O$ commutes with the cyclic shift $S(z)_n = z_{(n+1) \bmod 8}$.
\end{enumerate}
\end{definition}

\begin{definition}[LNAL generators]
The Light-Native Assembly Language (LNAL) comprises five primitive operators:
\[
\textsc{LISTEN},\quad \textsc{LOCK},\quad \textsc{BALANCE},\quad \textsc{FOLD},\quad \textsc{BRAID}.
\]
They respectively project into $H$, introduce balanced token-support pairs, re-centre skew, merge supports, and enact the unique trilinear interaction induced by $D=3$ spatial dimensions.
\end{definition}

\begin{theorem}[Operator uniqueness and minimality]
Every RS-legal operator $O$ factors as a finite composition of LNAL generators. No proper subset of the generator set suffices. Hence the LNAL calculus is unique.
\end{theorem}

\section{Normal form, coercivity, and meaning extraction}
Meaning extraction proceeds via the coercive projection method (CPM):
\begin{enumerate}
    \item \textbf{Alignment:} window signals to eight ticks and apply $P$.
    \item \textbf{Analysis:} project onto $\mathcal{B}_{\mathrm{axiom}}$ and enumerate candidate supports (size $\leq 4$).
    \item \textbf{Argmin:} choose the support minimising $J$ of the energy ratio.
    \item \textbf{Normal form:} reduce the chosen support to a canonical LNAL sequence.
\end{enumerate}
Coercivity of LNAL operators ensures that residual energy cannot grow under legal compositions. Strict convexity of $J$ guarantees the argmin is unique, inducing a confluent rewrite system for normal forms.

\begin{lemma}[Coercivity]
Each generator has minimum singular value at least $1$, so the measure ledger is non-decreasing under any LNAL composition.
\end{lemma}

\begin{lemma}[Argmin uniqueness]
Let $w \in H$. The optimisation
\[
\operatorname*{argmin}_{m \in \mathcal{M}} J\!\left(\frac{\|w - m\|_2}{\|w\|_2}\right),
\]
where $\mathcal{M}$ is the set of LNAL motifs generated from $\mathcal{B}_{\mathrm{axiom}}$, has a unique minimiser. The accompanying reduction rules yield a unique motif normal form.
\end{lemma}

\begin{theorem}[CPM coercivity]
For every $w \in H$,
\[
J\!\left(\frac{\|w - \Pi_{\mathcal{M}}(w)\|_2}{\|w\|_2}\right)
    \ge c_{\mathrm{coer}}\, \mathrm{Defect}(w),
\]
where $c_{\mathrm{coer}} = (C_{\mathrm{net}} C_{\mathrm{proj}} C_{\mathrm{eng}})^{-1} > 0$ and $\mathrm{Defect}$ is squared distance to the structured set generated by $\mathcal{B}_{\mathrm{axiom}}$ and LNAL motifs.
\end{theorem}

\section{Main theorem}

\begin{theorem}[Light Language completeness and uniqueness]
Let $H \subset \CC^{8}$ be the neutral subspace and let $\mathcal{B}_{\mathrm{axiom}} = \{\widehat{\psi}_k\}_{k=1}^{7}$ be the DFT-derived WTokens. Let $\mathsf{Ops} = \{\textsc{LISTEN}, \textsc{LOCK}, \textsc{BALANCE}, \textsc{FOLD}, \textsc{BRAID}\}$. Then:
\begin{enumerate}
    \item (\textbf{Completeness}) Every $w \in H$ has a unique decomposition $w = \sum_{k} c_k \widehat{\psi}_k$.
    \item (\textbf{Operator sufficiency}) Every RS-legal operator factors through $\mathsf{Ops}$.
    \item (\textbf{Normal form}) For every neutral signal, the CPM pipeline returns a unique LNAL normal form.
    \item (\textbf{Uniqueness}) If $(\mathcal{B}', \mathsf{Ops}')$ is any other complete $\tau_0$-neutral language, there exists a neutrality-preserving unitary $U$ such that $\mathcal{B}' = U\mathcal{B}_{\mathrm{axiom}}$ and $\mathsf{Ops}' = U \mathsf{Ops} U^{-1}$.
\end{enumerate}
Therefore the Light Language is the exclusive semantic language enforced by Recognition Science.
\end{theorem}
