\begin{abstract}
We present a companion to ``Morality as a Conservation Law'', extending the Recognition Science (RS) framework from feasibility to operation: fourteen \emph{virtues} are formalized as the complete set of admissible generators of ethical dynamics on the reciprocity-conserving manifold $\sigma=0$. Each generator is parameter-free and inherits the same invariants that calibrate physics: the unique convex cost $J(x)=\tfrac12(x+x^{-1})-1$ with $J''(1)=1$, the eight-tick cadence, and the $\varphi$-fixed scaling identities. The operational calculus couples four auditable quantities: (i)~reciprocity skew $\sigma$ (feasibility), (ii)~harm $\Delta S$ as the externalized action surcharge under least-action completion, (iii)~a forced axiology $V=\kappa I(A;E)-\mathcal C_J^*$ (mutual-information coupling minus $J$-induced curvature penalty; $\kappa$ fixed on a $\varphi$-tier), and (iv)~consent as the directional derivative sign $D_j V_i\ge 0$ along feasible directions. A lexicographic decision rule selects among $\sigma$-feasible actions without weights: minimize $\max\Delta S$, then maximize $\sum f(V)$ (curvature-normalized), then maximize the $\sigma$-graph spectral gap $\lambda_2$, breaking residual ties on the fixed $\varphi$-tier. We outline per-virtue invariants and audits, state a completeness/minimality program (DREAM), and list crisp falsifiers. If RS is the true physics, these generators constitute an \emph{inevitable} basis for moral action—auditable, falsifiable, and free of tunable knobs.
\end{abstract}

\noindent\textbf{Keywords:} Recognition Science; virtues as generators; reciprocity conservation $(\sigma=0)$; externalized action surcharge $(\Delta S)$; forced axiology $(V)$; lexicographic selection; parameter-free ethics; auditability; DREAM theorem.

\section{RS scaffold (recap)}
\label{sec:rs-scaffold}
This section summarizes the Recognition Science (RS) substrate on which the virtues calculus operates. Full derivations and proofs are given in the companion paper \textit{Morality as a Conservation Law}.

\paragraph{Unique convex cost and $\boldsymbol{\varphi}$ fixed point.}
RS fixes a single convex, symmetric cost on $\mathbb{R}_{>0}$,
\begin{equation}
J(x) \,=\, \tfrac12\big(x+x^{-1}\big)\,-\,1,\qquad x>0,\qquad J(1)=0,\quad J(x)=J(x^{-1}),\quad J''(1)=1.
\end{equation}
Self-similarity at this curvature normalization forces the golden ratio $\varphi$ as the unique positive fixed point of the induced scale recursion, $\varphi^2=\varphi+1$.

\paragraph{Eight-tick cadence and bridge identities.}
Time aggregates in minimal, cadence-invariant blocks of eight ticks (for $D=3$), realized by a ledger-compatible Hamiltonian walk on the 3-cube. Calibration to physical units is knob-free through dimensionless identities, e.g.
\begin{equation}
c=\frac{\ell_0}{\tau_0},\qquad \frac{c^3\,\lambda_{\mathrm{rec}}^2}{\hbar G}=\frac{1}{\pi}\ \ \Longleftrightarrow\ \ \lambda_{\mathrm{rec}}=\sqrt{\frac{\hbar G}{\pi c^3}}.
\end{equation}

\paragraph{Reciprocity skew and cycle minimality at $\boldsymbol{\sigma\!=\!0}$.}
Let $\sigma$ denote cycle-wise reciprocity skew (signed log-imbalance of bestowed vs received adjustments between parties). Strict convexity of $J$ at $x=1$ yields, for any $\varepsilon\neq 0$,
\begin{equation}
J(1+\varepsilon)+J(1-\varepsilon)\;>\; 2J(1)\,=\,0,
\end{equation}
so replacing any bidirectional pair $(1\!+\!\varepsilon,\,1\!-\!\varepsilon)$ by $(1,1)$ \emph{strictly lowers} the per-cycle action. Iterating this \emph{pairwise smoothing} over all imbalanced pairs yields:
\begin{quote}
\emph{A balanced cycle ($\sigma=0$) is action-minimizing; any persistent $\sigma\neq 0$ carries a strictly positive, avoidable action surplus.}
\end{quote}

\paragraph{Operational objects.}
On the $\sigma=0$ manifold, selection among feasible futures uses four audit quantities:
\begin{itemize}
  \item \textbf{Harm $\boldsymbol{\Delta S}$} (externalized action surcharge): for an act by $i$, the marginal increase in $j$'s required portion of the per-cycle action under the \emph{least-action completion} that preserves balance and $\sigma=0$.
  \item \textbf{Value $\boldsymbol{V}$} (forced axiology): uniquely fixed (up to a $\varphi$-tier scale $\kappa$) by gauge invariance, additivity, concavity, and curvature normalization,
  \begin{equation}
  V\;=\;\kappa\,I(A;E)\;-\;\mathcal{C}_J^{*},
  \end{equation}
  where $I(A;E)$ is agent–environment mutual information and $\mathcal{C}_J^{*}$ is the $J$-induced mechanical penalty under least-action completion.
  \item \textbf{Consent} as a \emph{directional derivative} on the feasible manifold: for $j$'s contemplated feasible direction $\xi$, agent $i$ consents iff
  \begin{equation}
  D_j V_i[\xi]\;=\;\left.\tfrac{d}{dt}\,V_i\big(\text{proj}_{\sigma=0}(t\xi)\big)\right|_{t=0}\ \ge\ 0.
  \end{equation}
  \item \textbf{Robustness} via the $\sigma$-graph spectral gap $\lambda_2$: the algebraic connectivity of the reciprocity network's Laplacian at $\sigma=0$, controlling worst-case surcharge amplification under bounded shocks.
\end{itemize}

These invariants and objects ground the virtues and the lexicographic selector in a parameter-free, auditable calculus shared with the companion derivations.

\section{MoralState, reciprocity, and feasibility}
\label{sec:moralstate}

\paragraph{MoralState.}
An agent-level ethical state is represented as
\[
\text{MoralState}\;s\;=\;(L,\,\{\mathcal E_i\}_{i\in\mathcal A},\,E;\,\text{valid}),
\]
where $L$ is a ledger snapshot (positive multipliers $x_e>0$ on directed bonds with site-by-site balance), $\{\mathcal E_i\}$ partitions bonds by agent/group, $E>0$ is an energy/audit budget, and \texttt{valid} asserts global admissibility (below). All updates are evaluated over one eight-tick cycle via the least-action completion operator $\Pi_{\mathrm{LA}}$ that rebalances non-acting bonds while keeping $\sigma=0$.

\paragraph{Reciprocity and curvature ($\boldsymbol{\kappa\equiv\sigma}$).}
For any ordered pair $(i,j)$ and cycle $C$, the pairwise skew is the signed log-imbalance
\[
\sigma_{ij}[C]\;:=\;\sum_{e\in i\to j}\!\ln x_e\;\; -\;\;\sum_{e\in j\to i}\!\ln x_e,\qquad \sigma_{ij}=-\sigma_{ji}.
\]
We write $\kappa$ synonymously for reciprocity skew (earlier ``curvature''): $\kappa\equiv\sigma$. Global skew is $\sigma[C]:=\sum_{i<j}|\sigma_{ij}[C]|$. Per-agent projections aggregate a domain's imbalances, e.g.\ $\kappa_i:=\sum_{j\ne i}|\sigma_{ij}|$ or, when needed, its signed components $\{\sigma_{ij}\}_j$.

\paragraph{Global admissibility and per-agent projections.}
A MoralState $s$ is \emph{globally admissible} iff its pre- and post-cycle ledgers satisfy
\begin{equation}
\sigma[C_{\text{pre}}]=0,\qquad \sigma[C_{\text{post}}]=0,\qquad E>0,
\end{equation}
with the post state obtained by applying the contemplated act on finitely many bonds and completing the remainder via $\Pi_{\mathrm{LA}}$. Per-agent projections $\Pi_i(L)$ restrict audit quantities (required-action portions, local skews) to $\mathcal E_i$ and feed $\Delta S(i\!\to\!j)$ and $V_i$.

\paragraph{Feasible transformations and virtues.}
Let $\mathcal{F}$ be the \emph{feasible manifold} at a $\sigma=0$ state: the set of nearby ledger states reachable in one eight-tick cycle that preserve site balance and $\sigma=0$ under $\Pi_{\mathrm{LA}}$. A \emph{feasible transformation} is a map $T:\mathcal{F}\to\mathcal{F}$ realized by acting on finitely many bonds and rebalancing the rest by least action.

\begin{definition}[Virtue, admissible generator on $\sigma=0$]\label{def:virtue}
A \emph{virtue} is an admissible generator $\mathcal{V}:\mathcal{F}\to\mathcal{F}$ that maps $\sigma$-feasible states to $\sigma$-feasible states in one eight-tick cycle and exposes the audit tuple $(\sigma,\Delta S, V, \lambda_2,\text{consent})$ computed with $\Pi_{\mathrm{LA}}$.
\end{definition}

In particular: (i)~\emph{feasibility} holds cycle-wise, (ii)~$\Delta S$ is the externalized surcharge under least-action completion, (iii)~$V=\kappa I(A;E)-\mathcal C_J^{*}$ is computed per agent with the fixed $\varphi$-tier, (iv)~\emph{consent} uses the directional derivative on $\mathcal{F}$, $D_j V_i[\xi]\ge 0$, and (v)~\emph{robustness} is summarized by the post-action $\sigma$-graph spectral gap $\lambda_2$. The next sections instantiate fourteen such generators and their invariants.

\section{Virtues calculus: definitions and invariants}
\label{sec:virtues-calculus}

\paragraph{Formal interface.}
A virtue exposes two maps: a \emph{transform} on the feasible manifold and an \emph{audit hook} for selection,
\begin{equation}
\mathcal{V}:\;\mathcal{F}\to\mathcal{F},\qquad
\mathsf{Audit}_{\mathcal{V}}:\;\mathcal{F}\to \mathsf{Report},
\end{equation}
with $\mathcal{F}$ the $\sigma=0$ manifold (Sec.~\ref{sec:moralstate}). The report is a structured tuple of invariants and scores used by the lexicographic selector.

\paragraph{Invariants enforced.}
Every virtue must satisfy the following \emph{hard} invariants per eight-tick cycle:
\begin{itemize}
  \item \textbf{Feasibility (reciprocity conservation):}
  \begin{equation}
  \sigma[C_{\mathrm{pre}}]=0,\qquad \sigma[C_{\mathrm{post}}]=0,\qquad L_{\mathrm{post}}=\Pi_{\mathrm{LA}}(\text{act}\mid L_{\mathrm{pre}}).
  \end{equation}
  Updates are applied on finitely many bonds; the least-action completion $\Pi_{\mathrm{LA}}$ rebalances the rest.
  \item \textbf{Energy constraints:} positivity of the audit budget and a per-cycle sustainability cap,
  \begin{equation}
  E_{\mathrm{post}}>0,\qquad \Delta E\;:=\;E_{\mathrm{pre}}-E_{\mathrm{post}}\;\le\;\frac{1}{\varphi}\,E_{\mathrm{pre}},
  \end{equation}
  unless explicitly tightened/neutralized by the specific virtue (Temperance saturates this bound).
  \item \textbf{$\boldsymbol{\varphi}$-tier normalization:} all numerical scales inherited from the forced axiology and cadence: the information scale $\kappa$ is fixed once (unit noiseless binary link $\mapsto$ one $\varphi$-tier), mechanical curvature is set by $J''(1)=1$; no new free weights are permitted.
\end{itemize}

\paragraph{Audit hooks and metrics.}
For a pre/post cycle $(C_{\mathrm{pre}},C_{\mathrm{post}})$ the audit hook returns
\begin{equation}
\mathsf{Report}\;=\;\big(\,\Sigma^{\mathrm{pre}},\,\Sigma^{\mathrm{post}},\,\Delta S,\,H,\,\Delta V,\,W,\,\lambda_2^{\mathrm{pre}},\,\lambda_2^{\mathrm{post}},\,\Delta\lambda_2,\,\mathsf{Consent}\,\big),
\end{equation}
where:
\begin{itemize}
  \item $\Sigma^{\mathrm{pre}}=\{\sigma_{ij}[C_{\mathrm{pre}}]\}_{i<j}$ and $\Sigma^{\mathrm{post}}=\{\sigma_{ij}[C_{\mathrm{post}}]\}_{i<j}$ are the \emph{$\sigma$ traces}.
  \item $\Delta S(i\!\to\!j)$ is the \emph{harm matrix} (externalized action surcharge under $\Pi_{\mathrm{LA}}$), with
  \begin{equation}
  H\;:=\;\max_{i,j}\,\Delta S(i\!\to\!j).
  \end{equation}
  \item $\Delta V_i:=V_i^{\mathrm{post}}-V_i^{\mathrm{pre}}$ and the welfare aggregate $W=\sum_i f(V_i)$ (curvature-normalized concave transform).
  \item $\lambda_2^{\mathrm{pre}}$ and $\lambda_2^{\mathrm{post}}$ are the $\sigma$-graph spectral gaps; $\Delta\lambda_2=\lambda_2^{\mathrm{post}}-\lambda_2^{\mathrm{pre}}$ summarizes robustness change.
  \item $\mathsf{Consent}$ is a table of derivative signs $\mathsf{C}(i\!\leftarrow\!j)=\big[\,D_j V_i[\xi]\ge 0\,\big]$ for the contemplated feasible directions $\xi$.
\end{itemize}

These metrics are computed \emph{after} applying the transform and projecting with $\Pi_{\mathrm{LA}}$. The lexicographic selector first filters on feasibility (traces zero), then minimizes $H$, maximizes $W$ on the tie set, then maximizes $\lambda_2$, and finally breaks any residual tie on the fixed $\varphi$-tier.

\section{Operational pieces (brief)}
\label{sec:operational}
This section summarizes the three core operators used by all virtues; full statements and proofs are in the companion \textit{Morality as a Conservation Law}.

\paragraph{Harm $\boldsymbol{\Delta S}$ (externalized action surcharge).}
Given an act by agent $i$ on a finite bond set, the \emph{required action} of $j$ is the minimum portion of per-cycle cost borne by $j$'s domain among all balanced, $\sigma=0$ completions (least-action projection). Harm is the marginal increase
\begin{equation}
\Delta S(i\!\to\! j)\;:=\;S_j^{*}[\text{act}]\,-\,S_j^{*}[\mathbf{1}].
\end{equation}
Properties: (P1)~\emph{gauge-invariance on the ledger} (unchanged under node-potential transforms and unit re-anchoring preserving $c=\ell_0/\tau_0$); (P2)~\emph{additivity} on independent subsystems; (P3)~\emph{composition} over the eight-tick cadence, $\Delta S$ sums per cycle.

\paragraph{Forced axiology $\boldsymbol{V}$ (MI–curvature).}
Axioms—gauge invariance, additivity on independent subsystems, concavity under coarse aggregation, and curvature normalization tied to $J''(1)=1$—force a unique cardinal form (up to a $\varphi$-tier scale $\kappa$):
\begin{equation}
V\;=\;\kappa\,I(A;E)\; -\; \mathcal{C}_J^{*},
\end{equation}
with $I(A;E)$ the agent–environment mutual information and $\mathcal{C}_J^{*}$ the $J$-induced mechanical penalty under least-action completion. The normalization $\kappa$ is fixed once by a canonical noiseless link; no free weights remain.

\paragraph{Consent (derivative sign on the $\boldsymbol{\sigma}$-manifold).}
For agent $j$'s contemplated feasible direction $\xi$ at a $\sigma=0$ state, agent $i$ \emph{consents} if the directional derivative of $i$'s value is nonnegative under least-action completion,
\begin{equation}
D_j V_i[\xi]\;=\;\left.\tfrac{d}{dt}\,V_i\big(\text{proj}_{\sigma=0}(t\xi)\big)\right|_{t=0}\ \ge\ 0.
\end{equation}
For non-competent agents, a conservative \emph{proxy consent} uses a lower-envelope axiology consistent with the four axioms; selection may temporarily override individual consent only under the lexicographic rule when no consent-respecting feasible alternative exists and minimax harm compels repair.

\begin{definition}[Consent on the feasible manifold]\label{def:consent}
Fix a $\sigma=0$ state and let $\mathcal{T}_s\mathcal{F}$ denote the set of feasible directions at $s$ (infinitesimal acts closed under $\Pi_{\mathrm{LA}}$). For $\xi\in\mathcal{T}_s\mathcal{F}$ attributed to agent $j$, agent $i$ \emph{consents} iff
\[
D_j V_i[\xi] \;\ge\; 0,\qquad\text{with }\ D_j V_i[\xi]:=\left.\tfrac{d}{dt}\,V_i\big(\text{proj}_{\sigma=0}(t\xi)\big)\right|_{t=0}.
\]
\end{definition}

\section{Lexicographic selector (decision principle)}
\label{sec:selector}
We select among $\sigma$-feasible actions using a parameter-free lexicographic order. Let $\mathcal{A}_\sigma$ be the nonempty set of actions that map a $\sigma=0$ state to a $\sigma=0$ state over one eight-tick cycle under $\Pi_{\mathrm{LA}}$. Define
\begin{equation}
H(a):=\max_{i,j}\Delta S(i\!\to\!j\mid a),\qquad
W(a):=\sum_i f\big(V_i\mid a\big),\qquad
R(a):=\lambda_2\big(L_\sigma\mid a\big),
\end{equation}
and let $\tau(a)$ be the fixed $\varphi$-tier tie-break rank.

\paragraph{Order of operations.}
\begin{equation}
\label{eq:lex}
\begin{aligned}
\mathcal{A}_1&:=\arg\min_{a\in\mathcal{A}_\sigma} H(a) \quad &&\text{(minimax max $\Delta S$)}\\
\mathcal{A}_2&:=\arg\max_{a\in\mathcal{A}_1} W(a) \quad &&\text{(maximize welfare)}\\
\mathcal{A}_3&:=\arg\max_{a\in\mathcal{A}_2} R(a) \quad &&\text{(maximize robustness)}\\
a^{*}&:=\arg\max_{a\in\mathcal{A}_3} \tau(a) \quad &&\text{($\varphi$-tier tie)}.
\end{aligned}
\end{equation}

\paragraph{Selector algorithm (concise).}
Given audited candidates $\{a_k\}$ with $\sigma$-feasible reports, compute $H(a_k), W(a_k), R(a_k)$:
\begin{enumerate}
  \item Filter: $\mathcal{A}_\sigma\gets\{a_k:\ \sigma\text{-traces post}=0\}$.
  \item Harm stage: $\mathcal{A}_1\gets\arg\min_{a\in\mathcal{A}_\sigma} H(a)$.
  \item Welfare stage: $\mathcal{A}_2\gets\arg\max_{a\in\mathcal{A}_1} W(a)$ \ (default $f(v)=v$).
  \item Robustness stage: $\mathcal{A}_3\gets\arg\max_{a\in\mathcal{A}_2} R(a)$.
  \item Tie-break: return $a^{*}\in\arg\max_{a\in\mathcal{A}_3} \tau(a)$.
\end{enumerate}

\paragraph{Why no weights appear.}
Introducing a scalar weight to blend $H$ and $W$ (or $W$ and $R$) would create a new, uncalibrated scale that is not fixed by the bridge or by $J''(1)=1$, violating parameter-free constraints (gauge, cadence, curvature). The lexicographic order respects these invariances: feasibility first, then harm minimax, then fixed-cardinal welfare, then robustness, then the same $\varphi$-tier arithmetic used elsewhere.

\begin{proposition}[Existence and selector optimality]\label{prop:selector}
If $\mathcal{A}_\sigma$ is finite or compact and $H,W,R$ are continuous on $\mathcal{A}_\sigma$, then \eqref{eq:lex} yields nonempty $\mathcal{A}_1,\mathcal{A}_2,\mathcal{A}_3$ and a well-defined $a^{*}$. Moreover, for any $a\in\mathcal{A}_\sigma$ either $a$ is infeasible (not in $\mathcal{A}_\sigma$) or
\begin{equation}
\big(H(a),\,-W(a)\big)\ \succeq_{\mathrm{lex}}\ \big(H(a^{*}),\,-W(a^{*})\big),
\end{equation}
with equality implying $R(a)\le R(a^{*})$, and if $R$ also ties then $\tau(a)\le \tau(a^{*})$.
\end{proposition}

\paragraph{Robustness margins.}
Let $\mathcal{U}_{\Delta S}$ and $\mathcal{U}_V$ be uncertainty sets for $\Delta S$ and $V$. Define
\begin{equation}
H^{\inf/\sup}(a):=\inf/\sup_{\theta\in\mathcal{U}_{\Delta S}} H_\theta(a),\qquad
W^{\inf/\sup}(a):=\inf/\sup_{\phi\in\mathcal{U}_V} W_\phi(a).
\end{equation}
Declare $a^{*}$ \emph{robustly certified} if
\begin{equation}
H^{\sup}(a^{*})\ <\ \min_{b\ne a^{*}} H^{\inf}(b),
\end{equation}
or, when the first margins tie,
\begin{equation}
H^{\sup}(a^{*})=\min_{b} H^{\inf}(b)\quad\text{and}\quad
W^{\inf}(a^{*})\ >\ \max_{b:\ H^{\inf}(b)=H^{\inf}(a^{*})} W^{\sup}(b),
\end{equation}
with the $R$ and $\tau$ refinements applied only within any remaining ties. If these margins fail, the audit returns \emph{indeterminate} and demands more measurement rather than weights.

\section{The fourteen virtues}
\label{sec:virtues}
Each virtue is presented with a consistent template: informal role; formal definition (pre/post constraints and parameters); enforced invariants ($\sigma=0$, energy, $\varphi$-tier); core theorems (e.g., $\sigma$-preservation, $\Delta S$ bounds, $V$ monotonicity, $\varphi$-optimality when applicable); Lean anchors; and a worked micro-example with audit outputs.

\subsection{Love — $\varphi$-ratio equilibration}
\textbf{Lean:} \texttt{IndisputableMonolith/Ethics/Virtues/Love.lean}

\textbf{Role.} Bilateral equilibration of local skew with $\varphi$-optimal energy split.

\textbf{Definition.} Act on two domains $(i,j)$; set pairwise skew to the average (unit in balanced normalization) and split energy $E_{\mathrm{tot}}$ as $E_i'=E_{\mathrm{tot}}/\varphi$, $E_j'=E_{\mathrm{tot}}/\varphi^2$; complete others by $\Pi_{\mathrm{LA}}$.

\textbf{Invariants.} $\sigma_{ij}^{\mathrm{post}}=0$; $E_i',E_j'>0$; $\varphi$-tier normalization (no free weights).

\textbf{Core theorems.} \texttt{love\_conserves\_skew}; \texttt{love\_reduces\_variance}; \texttt{love\_energy\_split\_is\_golden}.

\textbf{Micro-example.} Two domains with $E=(1.0,0.6)$; post: $E'=(0.618,0.382)$, $\sigma$-traces zero; $H=0.00$; $\Delta V=(+0.07,+0.05)$; $\Delta\lambda_2\approx+0.02$.

\subsection{Justice — accurate posting within eight-tick}
\textbf{Lean:} \texttt{IndisputableMonolith/Ethics/Virtues/Justice.lean}

\textbf{Role.} Ensure accurate ledger posting within the cadence.

\textbf{Definition.} A protocol with \texttt{posting} such that \texttt{accurate}: balances update exactly; \texttt{timely}: recording within 8 ticks.

\textbf{Invariants.} $\sigma^{\mathrm{post}}=0$; $E$ unchanged; cadence bound $\le 8$.

\textbf{Core theorems.} \texttt{justice\_preserves\_sigma\_zero}; \texttt{justice\_compositional}.

\textbf{Micro-example.} Single entry posted; $\sigma$-traces unchanged; $H=0.00$; $\Delta V=0$; timing certificate $\le 8$.

\subsection{Forgiveness — bounded skew transfer with energy cost}
\textbf{Lean:} \texttt{IndisputableMonolith/Ethics/Virtues/Forgiveness.lean}

\textbf{Role.} Relieve debtor skew with creditor energy payment under caps.

\textbf{Definition.} Transfer bounded $\Delta\kappa\le \kappa_{\max}$ from debtor to creditor; enforce $E_{\mathrm{cred}}^{\mathrm{post}}>0$ with explicit cost schedule.

\textbf{Invariants.} Global $\sigma=0$; $E_{\mathrm{cred}}^{\mathrm{post}}>0$; parameter-free cap expressed in ledger units.

\textbf{Core theorems.} \texttt{forgiveness\_conserves\_total\_skew}; \texttt{forgiveness\_bounded\_harm}; \texttt{forgiveness\_improves\_V\_debtor}.

\textbf{Micro-example.} Debtor $\kappa=+4$; transfer $\Delta\kappa=2$; $H=0.25$; $\Delta V_{\text{debtor}}=+0.12$; $\Delta V_{\text{cred}}=-0.03$.

\subsection{Wisdom — $\varphi$-discount horizon optimization}
\textbf{Lean:} \texttt{IndisputableMonolith/Ethics/Virtues/Wisdom.lean}

\textbf{Role.} Choose actions maximizing long-run value with $\varphi$-consistent discount (cadence-compliant).

\textbf{Definition.} Evaluate horizon policies with undiscounted eight-tick aggregation and fixed $\varphi$-tier normalization of $V$; select $\arg\max W$ on feasible set.

\textbf{Invariants.} $\sigma=0$ per cycle; no ad-hoc discount; $\varphi$-tier.

\textbf{Core theorems.} \texttt{wisdom\_respects\_cadence}; \texttt{wisdom\_monotone\_V}.

\textbf{Micro-example.} Two-step policy: higher immediate $\Delta V$ but worse $H$ is discarded; selected plan: $H=0.30\to0.10$, $\sum f(V)$ increases.

\subsection{Courage — high $|\nabla\sigma|$ moves, surcharge caps}
\textbf{Lean:} \texttt{IndisputableMonolith/Ethics/Virtues/Courage.lean}

\textbf{Role.} Permit decisive moves when skew gradients are high, under capped surcharge.

\section{Robustness and $\sigma$-graphs}
\label{sec:robustness}
At $\sigma=0$, linear response of the reciprocity network induces a weighted graph $G_\sigma$ with Laplacian $L_\sigma=D-W$ ($D_{ii}=\sum_j w_{ij}$). For a bounded, mean-zero shock $s$ with $\|s\|_2\le S$, the least-action response solves $L_\sigma u=s$, $u\perp\mathbf{1}$, and has Dirichlet energy $\mathcal{E}(u)=\tfrac12 s^\top L_\sigma^{+} s$.

\paragraph{Bound on worst surcharge.}
Node-wise surcharge obeys $\max_i \Delta S_i\le \mathcal{E}(u)\le \tfrac{S^2}{2\,\lambda_2(L_\sigma)}$. Thus larger $\lambda_2$ guarantees smaller worst-case $\Delta S$ under shocks.

\begin{lemma}[Robust-Preference]\label{lem:robust}
Among $\sigma$-neutral $W$-maximizers tied on present-cycle $H$, prefer the arrangement with larger $\lambda_2$: it minimizes the worst future surcharge across all bounded shocks.
\end{lemma}

This provides the third-stage tie-break in Sec.~\ref{sec:selector}.

\paragraph{Interpretation and policy.}
Higher algebraic connectivity means strain spreads quickly and thinly: fewer routes for skew to re-emerge and smaller worst bills per cycle under disturbances. When feasible designs tie on harm and welfare, policy should prefer those that increase $\lambda_2$ (e.g., add reciprocal ties that do not create skew).

\section{J-convexity and pairwise smoothing calculus}
\label{app:J}
Let
\[
J(x)=\tfrac12(x+x^{-1})-1,\qquad x>0,
\]
and write $x=e^{\alpha}$ so that $J(x)=\Phi(\alpha):=\cosh\alpha-1$ with $\Phi''(0)=1$.

\paragraph{Strict convexity near unity.}
For any $\varepsilon\ne 0$,
\begin{equation}
J(1+\varepsilon)+J(1-\varepsilon)\;=\;2\,\Phi\big(\ln(1+\varepsilon)\big)\;>\;0.
\end{equation}

\paragraph{Closed-cycle inequality.}
If $\{\alpha_k\}_{k=1}^m$ satisfies $\sum_k \alpha_k=0$ (product-one constraint on multipliers), then by Jensen's inequality for $\cosh$,
\begin{equation}
\sum_{k=1}^m \Phi(\alpha_k)\;\ge\; m\big(\cosh 0-1\big)\;=\;0,
\end{equation}
with equality iff $\alpha_k\equiv 0$. Thus the unit configuration minimizes the sum of per-bond costs on closed chains.

\paragraph{Pairwise smoothing step.}
For opposite-signed logs $\alpha>0$ and $-\beta<0$, replacing them by $(\alpha-t)$ and $-(\beta-t)$ for $t\in(0,\min\{\alpha,\beta\}]$ preserves the pairwise sum and strictly reduces $\Phi(\alpha)+\Phi(-\beta)$ by convexity. Iterating over all imbalanced pairs strictly decreases the total action until all pairs balance at unity.

\paragraph{Cycle minimality at $\boldsymbol{\sigma=0}$.}
Decomposing a cycle into bidirectional exchanges and applying the smoothing step yields: a per-cycle action is minimized subject to balance constraints iff all pairwise net skews vanish ($\sigma=0$). Any $\sigma\ne 0$ implies a strictly positive, avoidable action surplus.

From atomicity (T2: no concurrency) and $\delta$-units (T8) we obtain finiteness of in/out neighborhoods for each event $e$:
\begin{align}
&\textbf{AtomicFiniteOut:}\quad \{v\mid \texttt{evolves}\ e\ v\}\ \text{is finite},\\
&\textbf{UnitsFiniteIn:}\quad \{v\mid \texttt{evolves}\ v\ e\}\ \text{is finite}.
\end{align}
These yield a locally finite digraph representation (\textbf{LocalFinite\_from\_RS}).

\paragraph{Exactness $\Rightarrow$ balance (node conservation).}
Define a totalized edge value \texttt{edgeVal} and finitary inflow/outflow sums based on the local neighborhoods:
\begin{align}
\mathrm{inflowSum}\ f\ e&:=\sum_{v\in\mathrm{inNeigh}(e)} \mathrm{edgeVal}\ f\ (v,e),&
\mathrm{outflowSum}\ f\ e&:=\sum_{v\in\mathrm{outNeigh}(e)} \mathrm{edgeVal}\ f\ (e,v).
\end{align}
Under exactness on real cycles (closed-chain sums vanish), node-wise balance holds strongly:
\begin{equation}
\textbf{ConservationLawStrong:}\qquad \forall e,\ \mathrm{inflowSum}\ f\ e\;=\;\mathrm{outflowSum}\ f\ e.
\end{equation}

\paragraph{Exactness on real cycles.}
Represent a finite cycle as a vector $\langle e_0,\dots,e_{n-1}\rangle$ with successor map; define the closed-chain sum \texttt{closedChainSum\_real}. Well-foundedness of the reverse-evolution relation (MP forbids cycles on the empty type) implies no directed cycles; hence
\begin{equation}
\textbf{MP\_implies\_exactness\_real:}\qquad \forall\,\text{real cycles }C,\ \texttt{closedChainSum\_real}(f,C)=0.
\end{equation}
Combined with local finiteness this yields \textbf{exactness\_implies\_conservationStrong\_real}.

\paragraph{Divergence decomposition.}
On locally finite subgraphs, node divergence (outflow$-$inflow) decomposes into finitely many real cycles through the node; exactness kills each closed-chain sum, giving balance. (Lean skeleton: \texttt{divergence\_decomposes\_into\_cycles\_real}.)

\paragraph{Ledger equivalence.}
A balanced flow graph is equivalent to a double-entry ledger: there exists a ledger $L$ whose carrier is isomorphic to \texttt{Event} and whose debit/credit columns realize the same conservation law (\textbf{graph\_with\_balance\_is\_ledger}). The bridge theorem \textbf{mp\_implies\_conservation} packages the path
\[

\]

\section{No-Arbitrary-Discount: full proof and variants}
\label{app:nad}
Let $\mathcal H$ be the set of finite sequences $\mathbf h=(h_0,\dots,h_T)$ with $h_t\ge 0$ representing per-cycle worst surcharges $H(a_t)$ over an eight-tick cadence. A temporal aggregator is a map $\mathfrak A:\mathcal H\to\mathbb R$ used to rank repair paths.

\paragraph{Axioms.}
\begin{enumerate}
  \item[(D1)] \textbf{Gauge invariance (bridge):} joint re-anchoring $(\tau_0,\ell_0)\mapsto(s\tau_0,s\ell_0)$ preserving $c=\ell_0/\tau_0$ does not change rankings.
  \item[(D2)] \textbf{Cadence invariance:} rotation by multiples of eight ticks leaves $\mathfrak A$ unchanged; concatenation on independent horizons adds: $\mathfrak A(\mathbf h\Vert \mathbf k)=\mathfrak A(\mathbf h)+\mathfrak A(\mathbf k)$.
  \item[(D3)] \textbf{Separability and monotonicity:} $\mathfrak A$ is strictly increasing in each argument and continuous.
\end{enumerate}

\begin{theorem}[No-Arbitrary-Discount]\label{thm:nad}
Under (D1)–(D3), there is $\alpha>0$ such that $\mathfrak A(\mathbf h)=\alpha\,\sum_t h_t$ for all $\mathbf h\in\mathcal H$.
\end{theorem}

\begin{proof}
By (D2), all positions are equivalent under eight-tick rotations; define $\alpha:=\mathfrak A(e^{(t)})>0$ for unit vectors $e^{(t)}$. By concatenation and continuity, $\mathfrak A(\lambda \mathbf 1_n)=n\,\phi(\lambda)$ with $\phi$ continuous, increasing, and Cauchy-additive on $\mathbb R_{\ge 0}$; hence $\phi(\lambda)=\alpha\lambda$. For a general $\mathbf h=\sum_t h_t e^{(t)}$ use independence additivity across blocks and rotation invariance to obtain linearity: $\mathfrak A(\mathbf h)=\sum_t\mathfrak A(h_t e^{(t)})=\alpha\sum_t h_t$.
\end{proof}

\paragraph{Variants.}
\begin{itemize}
  \item \textbf{Weighted rotations forbidden:} any phase-dependent weights $(w_t)$ violate cadence invariance unless $w_t$ is constant across the eight positions and stable under concatenation (reducing to the sum case).
  \item \textbf{Exponential/hyperbolic discounts excluded:} any rate parameter introduces a time scale that rescales under (D1), breaking gauge invariance.
  \item \textbf{Segmented horizons:} if the horizon factorizes into independent segments, (D2) implies segment-wise additivity and the same linearity argument applies per segment.
  \item \textbf{Upper/lower envelopes:} uncertainty sets for $H$ produce interval-valued aggregators $\mathfrak A^{\inf/\sup}$; the same structure yields $\mathfrak A^{\inf/\sup}(\{h_t\})=\alpha\sum h_t$ with the same $\alpha$.
\end{itemize}
