{\Large Classical Inputs Used in the NS Proof}\\[4pt]
{\small Concise statement-level summary for referee convenience}
\end{center}

\vspace{4mm}
\noindent\textbf{A1. Critical $\varepsilon$--regularity (vorticity, $L^{3/2}$).}
For a suitable weak solution $(u,p)$ on $Q_{r_0}(x_0,t_0)$, if
\[
\Wc(x_0,t_0;r_0):=\frac1{r_0}\iint_{Q_{r_0}(x_0,t_0)} |\omega|^{3/2}\,dx\,dt\ \le\ \varepsilon_A,
\]
then
\[
\sup_{Q_{r_0/2}(x_0,t_0)} |\omega|\ \le\ \frac{C_A}{r_0^2}\,\Wc(x_0,t_0;r_0)^{2/3}.
\]
Notes: absorbed Caccioppoli + De Giorgi; parabolic scaling; standard in NS literature at critical scale.

\vspace{2mm}
\noindent\textbf{A2. Density--drop (De Giorgi improvement on smaller cylinders).}
There exist fixed $\vartheta\in(0,1/2)$, $c\in(0,1)$, $\eta_1>0$ such that if
$\Wc(0,0;1)\le \varepsilon_0+\eta$ with $0<\eta\le\eta_1$, then
\[\Wc(0,0;\vartheta)\ \le\ \varepsilon_0 + c\,\eta.\]
Notes: truncation $w=(|\omega|-\kappa_0)_+$ with $\kappa_0\sim\varepsilon_0^{2/3}$; absorbed Caccioppoli; ladder iteration.

\vspace{2mm}
\noindent\textbf{A3. Carleson characterization of $BMO^{-1}$.}
For $f\in\mathcal S'(\mathbb R^3)$,
\[\|f\|_{BMO^{-1}}\ \simeq\ \sup_{x\in\mathbb R^3,\ r>0}\Big(\frac{1}{|B_r|}\int_0^{r^2}\!\int_{B_r(x)} |e^{\nu\tau\Delta}f(y)|^2\,dy\,d\tau\Big)^{1/2}.\]
Notes: heat-flow square-function definition equivalent to the standard functional $BMO^{-1}$ norm.

\vspace{2mm}
\noindent\textbf{A4. Koch--Tataru small-data global theory in $BMO^{-1}$.}
There exists $\varepsilon_{\mathrm{SD}}>0$ such that if $\|u_0\|_{BMO^{-1}}\le\varepsilon_{\mathrm{SD}}$, then there is a unique global mild solution $u$ with $u\in X$ (KT space), smooth for $t>0$.
Notes: includes bilinear estimate in the $X$ space and continuity of the $BMO^{-1}$ norm on short times.

\vspace{2mm}
\noindent\textbf{A5. Uniqueness: backward Carleman + forward energy.}
If $u,v$ solve NS on $\mathbb R^3\times[t_1,t_2]$, with $v$ smooth and $u$ suitable, and $u(\cdot,t_0)=v(\cdot,t_0)$ for some $t_0\in(t_1,t_2)$, then $u\equiv v$ on $\mathbb R^3\times[t_1,t_2]$.
Notes: local backward uniqueness from a parabolic Carleman estimate; forward uniqueness by standard $L^2$ energy.

\vspace{2mm}
\noindent\textbf{A6. Compactness and critical-element extraction.}
Given a sequence of suitable solutions with uniform local $L^3$ bounds (and pressure in $L^{3/2}$), there is strong $L^3_{\mathrm{loc}}$ compactness on interior cylinders; $\Wc$ is lower semicontinuous under this convergence. These yield existence of a nontrivial ancient critical element at a minimal profile level.

\vspace{2mm}
\noindent\textbf{Auxiliary local embeddings (derived internally).}
\begin{itemize}
  \item (LE1) \emph{Local $BMO^{-1}\!\to L^3$ on a slice:} for every $t$ and ball $B_r(x)$,
  $\displaystyle \|u(\cdot,t)\|_{L^3(B_r(x))}\ \lesssim\ r^{1/2}\,\|u(\cdot,t)\|_{BMO^{-1}}$.
  \item (LE2) \emph{Local $L^3\!\to L^{3/2}$ for vorticity:} for every $t$ and ball $B_r(x)$,
  $\displaystyle \|\omega(\cdot,t)\|_{L^{3/2}(B_r(x))}\ \lesssim\ r^{-1/2}\,\|u(\cdot,t)\|_{L^3(B_{2r}(x))}$.
\end{itemize}
Notes: consequences of A3 (Carleson/BMO$^{-1}$), heat-kernel smoothing, Calderón–Zygmund, and local Poincaré; no extra external inputs are required.
